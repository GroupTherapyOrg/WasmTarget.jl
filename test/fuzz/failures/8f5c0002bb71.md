```julia
using WasmTarget
include(joinpath("test", "fuzz", "harness.jl")); using .FuzzHarness
struct BPointG; x::Float64; y::Float64; end
mk(x::Float64) = BPointG(x, -x)
fget1(p::BPointG)::Float64 = getfield(p, 1)
bytes = WasmTarget.compile_multi([(mk, (Float64,), "mk"), (fget1, (BPointG,), "fget1")]; validate=true)
status, results = FuzzHarness.run_driver_batch(bytes, """
const { instance } = await WebAssembly.instantiate(bytes, { Math: { pow: Math.pow } });
const ex = instance.exports;
try { return [{ ok: ex["fget1"](ex["mk"](2.5)) }]; } catch (e) { return [{ trap: String(e.message) }] }
"""; ninputs = 1)
r = results[1]
haskey(r, "ok") && r["ok"] == 2.5 || error("WasmTarget gap: getfield(p, 1) on a JS-crossed ref => " * string(r))
```

## Diagnostic
```
exports["fget1"](exports["mk"](2.5)) traps `unreachable`, while the identical call sequence with `p.x`-form access returns correctly, and calling fget1(mk(x)) INSIDE wasm also works. Integer-index getfield lowering apparently takes a dynamic path that breaks for externalized refs. FuzzBridge works around it by always using x.name / x[i] forms.
```
